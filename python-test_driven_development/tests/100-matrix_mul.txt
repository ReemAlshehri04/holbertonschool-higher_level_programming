Test cases for matrix_mul function
====================================

>>> matrix_mul = __import__('100-matrix_mul').matrix_mul

# Basic tests from example
>>> matrix_mul([[1, 2], [3, 4]], [[1, 2], [3, 4]])
[[7, 10], [15, 22]]

>>> matrix_mul([[1, 2]], [[3, 4], [5, 6]])
[[13, 16]]

# Edge cases
>>> matrix_mul([[1]], [[2]])
[[2]]

>>> matrix_mul([[1, 2, 3]], [[4], [5], [6]])
[[32]]

>>> matrix_mul([[4], [5], [6]], [[1, 2, 3]])
[[4, 8, 12], [5, 10, 15], [6, 12, 18]]

# Float tests with CORRECT calculation
>>> matrix_mul([[1.5, 2.5], [3.5, 4.5]], [[2.0, 3.0], [4.0, 5.0]])
[[13.0, 17.0], [25.0, 33.0]]

# 2x2 * 2x3 = 2x3 (mathematically correct)
>>> matrix_mul([[1, 2], [3, 4]], [[1, 2, 3], [4, 5, 6]])
[[9, 12, 15], [19, 26, 33]]

# This CANNOT be multiplied! 2x2 * 3x2
>>> try:
